
import java.io.Console;

public class App {
    static Console c = System.console();
    static BinaryChoice binaryChoice = new BinaryChoice();
    static Person person = new Person();

    public static void chooseGender() {

        System.out.println("What gender would you like to be? Female, Male or Non-binary");
        String genderResponse = c.readLine();
        if (genderResponse.equalsIgnoreCase("Female")) {
            person.setGender(1);
        } else if (genderResponse.equalsIgnoreCase("Male")) {
            person.setGender(2);
        } else if (genderResponse.equalsIgnoreCase("Non-binary")) {
            person.setGender(3);
        } else {
            System.out.println("Please choose a gender listed"); // failsafe
            chooseGender();
        }

    }

    public static void homeworkButShorter() {
        System.out.println("You come home from 1st grade. Your mom tells you 'You need to do your math homework!' ");
        System.out.println("Do you do your math homework?");

        BinaryChoice.choice(person, "Yes", "No", 5, -5, "You begin doing your math homework.",
                "You get a spanking for not studying. Your mom then forces you to study math.",
                BinaryChoice.REP_MODIFY);

        person.setAge(person.getAge() + 1);
        System.out.println("You've aged and are now " + person.getAge() + " years old.");
    }

    public static void homework() {
        System.out.println("You come home from 1st grade. Your mom tells you 'You need to do your math homework!' ");
        System.out.println("Do you do your math homework?");
        String homeworkResponse = c.readLine();

        if (homeworkResponse.equalsIgnoreCase("Yes")) {
            System.out.println("You begin doing your math homework");
            TerminalColor.printLine("You gained reputation points +5; you now have " + person.getReputation(),
                    TerminalColor.GREEN_BRIGHT);
            System.out.println("You aged +1; you are now 8 years old!");
            person.setAge(person.getAge() + 1);
            person.setReputation(person.getReputation() + 5);
        } else if (homeworkResponse.equalsIgnoreCase("no")) {
            System.out.println("You get a spanking for not studying. Your mom then forces you to study your math.");
            TerminalColor.printLine("You lose reputation points -5; you now have " + person.getReputation(),
                    TerminalColor.RED_BRIGHT);
            System.out.println("You aged +1; you are now 8 years old!");
            person.setAge(person.getAge() + 1);
            person.setReputation(person.getReputation() - 5);

        } else {
            System.out.println("Please choose yes or no");// failsafe
            homework();
        }
    }

    public static void pencil() {
        System.out.println("You head to class but your pencil breaks.");
        System.out.println("Do you steal your classmates pencil?");
        String pencilResponse = c.readLine();
        if (pencilResponse.equalsIgnoreCase("Yes")) {
            person.setReputation(person.getReputation() - 5);
            System.out.println("Your classmate sees you and tells the teacher");
            TerminalColor.printLine("You lose reputation points -5; you now have " + person.getReputation(),
                    TerminalColor.RED_BRIGHT);
            System.out.println("You aged +1; you are now 9 years old!");
            person.setAge(person.getAge() + 1);
        } else if (pencilResponse.equalsIgnoreCase("No")) {
            person.setReputation(person.getReputation() + 0);
            System.out.println("You ask the teacher and the teacher gives you a pencil");
            TerminalColor.printLine("You didn't gain any reputation; you still have " + person.getReputation(),
                    TerminalColor.BLUE_BRIGHT);
            System.out.println("You aged +1; you are now 9 years old!");
            person.setAge(person.getAge() + 1);

        } else {
            System.out.println("Please choose yes or no"); // failsafe
            pencil();
        }
    }

    public static void tag() {
        System.out.println("A kid comes to you and tags you");
        System.out.println("Please choose STRANGER DANGER! or COME BACK HERE"); // failsafe
        String tagResponse = c.readLine();

        if (tagResponse.equalsIgnoreCase("STRANGER DANGER!")) {
            person.setReputation(person.getReputation() - 5);
            System.out.println("You punch the kid for no apparent reason");
            TerminalColor.printLine("You lose reputation points -5; you now have " + person.getReputation(),
                    TerminalColor.RED_BRIGHT);
            System.out.println("You aged +1; you are now 10 years old!");
            person.setAge(person.getAge() + 1);
        } else if (tagResponse.equalsIgnoreCase("COME BACK HERE")) {
            person.setReputation(person.getReputation() + 5);
            System.out.println("You befriend the kid");
            TerminalColor.printLine("You gained reputation points +5; you now have " + person.getReputation(),
                    TerminalColor.GREEN_BRIGHT);
            System.out.println("You aged +1; you are now 10 years old!");
            person.setAge(person.getAge() + 1);
        } else {
            System.out.println("please choose STANGER DANGER or COME BACK HERE"); // failsafe
            tag();
        }
    }

    public static void ditch() {
        System.out.println("A classmate asks if you wanna ditch class with them to play games");
        System.out.println(" Please choose; Schools boring! lets go!!! or No thanks, we got a test today!");
        String ditchResponse = c.readLine();
        if (ditchResponse.equalsIgnoreCase("Schools boring! lets go!!!")) {
            person.setReputation(person.getReputation() - 20);
            System.out.println("Your mom spots you leaving school. She drags you by the ear back into school");
            TerminalColor.printLine("You lose reputation points -20; you now have " + person.getReputation(),
                    TerminalColor.RED_BRIGHT);
            System.out.println("You aged +1; you are now 11 years old!");
            person.setAge(person.getAge() + -20);
        } else if (ditchResponse.equalsIgnoreCase("No thanks, we got a test today!")) {
            person.setReputation(person.getReputation() + 20);
            System.out.println("");
            TerminalColor.printLine("You gained reputation points + 20; you now have " + person.getReputation(),
                    TerminalColor.GREEN_BRIGHT);
            System.out.println("You aged +1; you are now 11 years old!");
            person.setAge(person.getAge() + 20);
        } else {
            System.out.println("please choose STANGER DANGER or COME BACK HERE"); // failsafe
            tag();
        }
    }
